Cost-Efficient RAG Application COMPLETE IMPLEMENTATION GUIDE

1. PROJECT OVERVIEW

This project implements a cost-efficient Retrieval-Augmented Generation (RAG) application using FAISS (Facebook AI Similarity Search) as the vector store with SQLite for metadata storage. This combination provides a low-cost alternative to fully managed vector databases while maintaining good performance.

Why FAISS + SQLite?

FAISS: Industry-standard, highly optimized vector similarity search SQLite: Zero-configuration, embedded database for metadata Cost: No ongoing infrastructure costs (runs on local/cheap VPS) Performance: Excellent for up to 10M vectors Simplicity: Easy to set up and maintain

def ingest_file(self, file_path: str, metadata: Optional[Dict[str, Any]] = None) -> int:

Ingest a single file

Args:

file_path: Path to the file

metadata: Additional metadata

Returns:

Number of chunks ingested

}

if metadata:

base_metadata.update(metadata)

def answer(

self,

query: str,

k: Optional[int] = None,

filter_metadata: Optional[Dict[str, Any]] = None

) -> Dict[str, Any]:

"""

Answer a query using RAG

Args:

query: User query

k: Number of chunks to retrieve

filter_metadata: Optional metadata filter Returns:

Dictionary with answer and metadata

"""

start_time = time.time()

# Retrieve relevant chunks

retrieval_start = time.time()

chunks = self.retriever.retrieve(query, k=k, filter_metadata=filter_metadata)

retrieval_time = time.time() - retrieval_start

# Format context

context = self.retriever.format_context(chunks)

# Generate answer

answer_start = time.time()

answer, tokens_used = self._generate_answer(query, context, chunks)

answer_time = time.time() - answer_start total_time = time.time() - start_time

# Log metrics

logger.info(f"Query: {query[:50]}...") logger.info(f"Retrieved {len(chunks)} chunks in

{retrieval_time*1000:.2f}ms")

logger.info(f"Answer generated in {answer_time*1000:.2f}ms") logger.info(f"Total time: {total_time*1000:.2f}ms") logger.info(f"Tokens used: {tokens_used}") return {

"query": query,

"answer": answer,

"context": context,

"chunks": chunks,

"metrics": {

"retrieval_time_ms": retrieval_time * 1000,

"answer_time_ms": answer_time * 1000,

"total_time_ms": total_time * 1000,

"chunk_count": len(chunks),

"tokens_used": tokens_used

}

}

def _generate_answer(

self,

query: str,

context: str,

chunks: List[Dict[str, Any]]

) -> tuple[str, int]:

"""

Generate an answer using the LLM

Args:

query: User query

context: Formatted context

chunks: Retrieved chunks

Returns:

Tuple of (answer, tokens_used)

"""

if not self.openai_client:

return self._generate_fallback_answer(query, context), 0

if not chunks:

return self._handle_no_context(query), 0

Context:

{context}

Question: {query}

Instructions:

1. Answer based ONLY on the context provided.

2. If the context doesn't contain the answer, say "I don't have enough information to answer this question."

3. Cite the sources you use (e.g., [1], [2]).

messages=[

],

def _compute_retrieval_metrics(

self,

retrieved_chunks: List[Dict[str, Any]],

relevant_chunk_ids: List[str]

) -> Dict[str, float]:

"""Compute retrieval metrics"""

# Get retrieved IDs

retrieved_ids = [c["id"] for c in retrieved_chunks]

# Hit rate (recall@k)

hits = sum(1 for rid in retrieved_ids if rid in relevant_chunk_ids)

hit_rate = hits / min(len(relevant_chunk_ids), len(retrieved_ids)) if relevant_chunk_ids else 0

# Precision@k
